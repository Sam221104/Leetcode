select stu.student_id, stu.student_name, su.subject_name,count(e.student_id) as attended_exams from Students stu cross join subjects su left join Examinations e   on stu.student_id = e.student_id and e.subject_name = su.subject_name group by stu.student_id, stu.student_name, su.subject_name order by stu.student_id
